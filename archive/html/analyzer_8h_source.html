407  void show_attr(attributes_t *attr) {
408  int have_nexthop = 0;
409  printf("ATTRIBUTES :\n");
410 
411  if (attr != NULL) {
412  printf(" ATTR_LEN : %d\n", attr->len);
413 
414  if ((attr->flag & ATTR_FLAG_BIT(BGP_ATTR_ORIGIN)) != 0) printf(" ORIGIN : %d\n", attr->origin);
415  else printf(" ORIGIN : N/A\n");
416 
417  if ((attr->flag & ATTR_FLAG_BIT(BGP_ATTR_AS_PATH)) != 0) printf(" ASPATH : %s\n", attr->aspath->str);
418  else printf(" ASPATH : N/A\n");
419 
420  printf(" NEXT_HOP : ");
421  if ((attr->flag & ATTR_FLAG_BIT(BGP_ATTR_NEXT_HOP)) != 0) {
422  have_nexthop = 1;
423  printf("%s", inet_ntoa(attr->nexthop));
424  }
425 
426 #ifdef BGPDUMP_HAVE_IPV6
427  if ((attr->flag & ATTR_FLAG_BIT(BGP_ATTR_MP_REACH_NLRI)) &&
428  MP_IPV6_ANNOUNCE(attr->mp_info) != NULL) {
429  char addr[INET6_ADDRSTRLEN];
430  struct mp_nlri *mp_nlri = MP_IPV6_ANNOUNCE(attr->mp_info);
431  u_int8_t len = mp_nlri->nexthop_len;
432 
433  if (have_nexthop)
434  printf(" ");
435 
436  have_nexthop = 1;
437  printf("%s", inet_ntop(AF_INET6, &mp_nlri->nexthop, addr, sizeof(addr)));
438  if (len == 32)
439  printf(" %s", inet_ntop(AF_INET6, &mp_nlri->nexthop_local, addr, sizeof(addr)));
440  }
441 #endif
442 
443  printf(have_nexthop ? "\n" : "N/A\n");
444 
445  if ((attr->flag & ATTR_FLAG_BIT(BGP_ATTR_MULTI_EXIT_DISC)) != 0) printf(" MED : %d\n", attr->med);
446  else printf(" MED : N/A\n");
447 
448  if ((attr->flag & ATTR_FLAG_BIT(BGP_ATTR_LOCAL_PREF)) != 0) printf(" LOCAL_PREF : %d\n", attr->local_pref);
449  else printf(" LOCAL_PREF : N/A\n");
450 
451  if ((attr->flag & ATTR_FLAG_BIT(BGP_ATTR_ATOMIC_AGGREGATE)) != 0) printf(" ATOMIC_AGREG : Present\n");
452  else printf(" ATOMIC_AGREG : N/A\n");
453 
454  if ((attr->flag & ATTR_FLAG_BIT(BGP_ATTR_AGGREGATOR)) != 0) printf(" AGGREGATOR : %s AS%u\n", inet_ntoa(attr->aggregator_addr), attr->aggregator_as);
455  else printf(" AGGREGATOR : N/A\n");
