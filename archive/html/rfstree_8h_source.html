520

522 
523  size_t qArr[W - U + 1][K]; // per-stage memory requirement
524 
525  size_t rArr[W - U + 1][K]; // level selected to be expanded in the previous iteration
526 
527  //initialize pArr, qArr and rArr
528  for (int i = 0; i < W - U + 1; ++i) {
529 
530  for (int j = 0; j < K; ++j) {
531 
532  pArr[i][j] = std::numeric_limits<size_t>::max();
533 
534  qArr[i][j] = std::numeric_limits<size_t>::max();
535 
536  rArr[i][j] = std::numeric_limits<size_t>::max();
537  }
538  }
539 
540  // pArr[i][0]
541  // pArr[i][0] = qArr[i][0] = 2^i, rArr[i][0] = W + 1. That is, use expansion level 0 to represent levels 0, 1, ..., i
542  for (int i = 1; i < W - U + 1; ++i) {
543 
544  pArr[i][0] = _rbt->getLevelNodeNum(0) * static_cast<size_t>(pow(2, i));
545 
546  qArr[i][0] = pArr[i][0];
547 
548  rArr[i][0] = W - U + 1; // no previous iteration
549  }
550 
551  // pArr[i][i] = pArr[i - 1][i - 1] + levelnodenum[i], qArr[i][i] = max{qArr[i - 1][i - 1], levelnodenum[i]}
552  pArr[0][0] = _rbt->getLevelNodeNum(0);
553 
554  qArr[0][0] = pArr[0][0];

565  }
